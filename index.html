    <div id="quiz-creator" style="display: block;">

        <textarea id="input-text" placeholder="Paste your text here..."></textarea>

        <button onclick="processText()">Process Text</button>

        <div id="text-display"></div>

        <button onclick="generateShareCode()" style="display:none;" id="share-button">Generate Share Code</button>

        <input type="text" id="share-code" readonly style="display:none;">

    <div id="myModal" class="modal">

        <div class="modal-content">

            <span class="close">&times;</span>

            <h2>Enter wrong answers</h2>

            <p>Correct answer: <span id="correct-answer"></span></p>

            <input type="text" id="wrong1" placeholder="Wrong answer 1">

            <input type="text" id="wrong2" placeholder="Wrong answer 2">

            <input type="text" id="wrong3" placeholder="Wrong answer 3">

            <button onclick="submitWrongAnswers()">Submit</button>

        </div>

    </div>


    <script>

        let correctAnswers = {};

        let questionCount = 0;

        let currentSpan = null;

        let currentWord = null;

        let currentIndex = null;

        let quizData = {};


        function processText() {

            const text = document.getElementById('input-text').value;

            const words = text.split(/\s+/);

            const displayDiv = document.getElementById('text-display');

            displayDiv.innerHTML = '';


            words.forEach((word, index) => {

                const span = document.createElement('span');

                span.textContent = word + ' ';

                span.onclick = () => openModal(span, word, index);

                displayDiv.appendChild(span);

            });


            document.getElementById('share-button').style.display = 'inline-block';

        }


        function openModal(span, word, index) {

            currentSpan = span;

            currentWord = word;

            currentIndex = index;

            document.getElementById('correct-answer').textContent = word;

            document.getElementById('myModal').style.display = 'block';

            document.getElementById('wrong1').value = '';

            document.getElementById('wrong2').value = '';

            document.getElementById('wrong3').value = '';

        }


        function submitWrongAnswers() {

            const wrong1 = document.getElementById('wrong1').value;

            const wrong2 = document.getElementById('wrong2').value;

            const wrong3 = document.getElementById('wrong3').value;


            if (wrong1 && wrong2 && wrong3) {

                createGap(currentSpan, currentWord, currentIndex, [wrong1, wrong2, wrong3]);

                document.getElementById('myModal').style.display = 'none';

            } else {

                alert('Please enter all three wrong answers.');

            }

        }


        function createGap(span, word, index, wrongAnswers) {

            questionCount++;

            const selectId = `q${questionCount}`;

            correctAnswers[selectId] = word;


            const select = document.createElement('select');

            select.id = selectId;


            const options = [word, ...wrongAnswers];

            shuffleArray(options);


            options.unshift('Choose...');


            options.forEach(option => {

                const optionElement = document.createElement('option');

                optionElement.value = option;

                optionElement.textContent = option;

                select.appendChild(optionElement);

            });


            span.parentNode.replaceChild(select, span);


            // Store quiz data

            quizData[selectId] = {

                correct: word,

                options: options.slice(1)  // Exclude 'Choose...'

            };

        }


        function shuffleArray(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

        }


        function checkAnswers() {

            let score = 0;

            let total = Object.keys(correctAnswers).length;


            for (let question in correctAnswers) {

                if (document.getElementById(question).value === correctAnswers[question]) {

                    score++;

                }

            }


            document.getElementById("result").innerHTML = `Your score: ${score} out of ${total}`;

        }


        function generateShareCode() {

            const quizJSON = JSON.stringify(quizData);

            const shareCode = btoa(quizJSON); // Encode the JSON as base64

            

            document.getElementById('share-code').value = shareCode;

            document.getElementById('share-code').style.display = 'block';

        }


        function loadSharedQuiz() {

            const shareCode = prompt("Enter the share code for the quiz:");

            if (shareCode) {

                try {

                    const quizJSON = atob(shareCode); // Decode the base64 string

                    const loadedQuizData = JSON.parse(quizJSON);


                    document.getElementById('quiz-creator').style.display = 'none';

                    document.getElementById('quiz-taker').style.display = 'block';


                    const quizContent = document.getElementById('quiz-content');

                    quizContent.innerHTML = '';


                    for (let questionId in loadedQuizData) {

                        const questionData = loadedQuizData[questionId];

                        const select = document.createElement('select');

                        select.id = questionId;


                        const options = ['Choose...', ...questionData.options];

                        options.forEach(option => {

                            const optionElement = document.createElement('option');

                            optionElement.value = option;

                            optionElement.textContent = option;

                            select.appendChild(optionElement);

                        });


                        quizContent.appendChild(select);

                        quizContent.appendChild(document.createTextNode(' '));


                        correctAnswers[questionId] = questionData.correct;

                    }

                } catch (error) {

                    alert("Invalid share code. Please try again.");

                }

            }

        }


        // Close modal when clicking on <span> (x)

        document.getElementsByClassName("close")[0].onclick = function() {

            document.getElementById('myModal').style.display = "none";

        }


        // Close modal when clicking outside of it

        window.onclick = function(event) {

            if (event.target == document.getElementById('myModal')) {

                document.getElementById('myModal').style.display = "none";

            }

        }
